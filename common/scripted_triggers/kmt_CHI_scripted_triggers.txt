	OR = {
		is_in_faction = no
		is_in_faction_with = CHI
		AND = {
			is_in_faction = yes
			NOT = {
				is_in_faction_with = CHI
			}
		}
	}
	NOT = {
		any_country = {
			is_in_faction_with = PREV
			has_war_with = CHI
		}
	}
}
