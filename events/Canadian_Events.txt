########################### Canadian Prime Minister meets Hitler ###########################
country_event = {
		
	id = canada.4
	title = canada.4.t
	desc = canada.4.d
	picture = GFX_report_event_hitler_meets_troops

	trigger = {
		tag = CAN
		date > 1937.5.1	
		NOT = { has_war_with = GER }
	}
	
	mean_time_to_happen = { days = 10 }
	
	fire_only_once = yes

	option = {
		name = canada.4.a

}
